%% v1.4 NOTE:

    $\displaystyle
      \zeta(s)
      = \sum_{n=1}^\infty \frac{1}{n^s}
      = \frac{1}{\Gamma(s)} \int_0^\infty \frac{x^{s-1}}{e^x - 1} \, dx
      \quad (\Re(s) > 1)
    $
  };
