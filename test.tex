\subsection*{Riemann Zeta Function}
\begin{tikzpicture}
  \node[
    draw=DIC161s,
    line width=1.5pt,
    fill=DIC161s!5,
    rounded corners=2ex,
    inner sep=1.5em
  ] {
%% v1.4 NOTE:
%% CMYK値: 0 0.512 0.8 0.2
%% 色: DIC161s!80!black
%% 理由: 非比例混色（K成分0.2が追加）→ 正しいフォールバック
    \color{DIC161s!80!black}
    $\displaystyle
      \zeta(s)
      = \sum_{n=1}^\infty \frac{1}{n^s}
      = \frac{1}{\Gamma(s)} \int_0^\infty \frac{x^{s-1}}{e^x - 1} \, dx
      \quad (\Re(s) > 1)
    $
  };
\end{tikzpicture}
